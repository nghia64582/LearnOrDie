- User:
    + id: int(11) auto_increment primary key
    + username: varchar(50)
    + password: varchar(50)
    + email: varchar(50)
    + created_at: datetime
    + updated_at: datetime
    + last_login: datetime
    + status: enum('active', 'inactive', 'banned')

- Role:
    + id: int(11) auto_increment primary key
    + name: varchar(50) ("ROLE_ADMIN", "ROLE_USER")
    + description: text
    + created_at: datetime
    + updated_at: datetime

- Product:
    + id: int(11) auto_increment primary key
    + name: varchar(100)
    + description: text
    + price: decimal(10,2)
    + stock: int(11)
    + category_id: int(11) foreign key references Category(id)
    + created_at: datetime
    + updated_at: datetime
    + status: enum('active', 'inactive', 'deleted')

- Category:
    + id: int(11) auto_increment primary key
    + name: varchar(50)
    + description: text
    + created_at: datetime
    + updated_at: datetime

- Order:
    + id: int(11) auto_increment primary key
    + user_id: int(11) foreign key references User(id)
    + total_price: decimal(10,2)
    + status: enum('pending', 'completed', 'cancelled')
    + created_at: datetime
    + updated_at: datetime

- UserReaction:
    + id: int(11) auto_increment primary key
    + user_id: int(11) foreign key references User(id)
    + product_id: int(11) foreign key references Product(id)
    + reaction_type: enum('like', 'show_phone', 'show_email')
    + created_at: datetime

- UserNotification:
    + id: int(11) auto_increment primary key
    + user_id: int(11) foreign key references User(id)
    + notification_type: enum('email', 'sms', 'push')
    + message: text
    + status: enum('unread', 'read')
    + created_at: datetime
    + updated_at: datetime

- Message:
    + id: int(11) auto_increment primary key
    + sender_id: int(11) foreign key references User(id)
    + receiver_id: int(11) foreign key references User(id)
    + content: text
    + created_at: datetime
    + status: enum('sent', 'received', 'read')
